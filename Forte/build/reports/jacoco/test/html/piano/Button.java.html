package piano;

import processing.core.PImage;
import processing.core.PApplet;

public class Button {
    private int x;
    private int y;
    private PImage[] buttonImages;
    private PImage currButton;
    
    public Button(int x, int y, PImage[] buttonImages) {
        this.x = x;
        this.y = y;
        this.buttonImages = buttonImages;
        this.currButton = buttonImages[0]; // default showing play button
    }

    public void draw(PApplet app) {
        // Handling graphics
        app.image(this.currButton, this.x, this.y);
    }

    public void update() {
        try {
            if(this.currButton.equals(this.buttonImages[0])) {
                this.currButton = this.buttonImages[1];
            } else {
                this.currButton = this.buttonImages[0];
            }
        } catch (NullPointerException e) {
            this.currButton = null;
        }
    }

    public void reset() {
        this.currButton = this.buttonImages[0];
    }

    public PImage getCurr() { // for testing only
        return this.currButton;
    }
}
